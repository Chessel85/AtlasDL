//Get admin levels 2 to 6 from Europe 
//Starting with countries admin level 2 
//Exclude France and Russia and United States 
layerName=countries
layerGeometry=polygons
createLayer=yes
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=2]
geodeskAreaFilter=(-15.0,34.0,60.0,80.0)
excludeByTag=[name:en=France]
excludeByTag=[name:en='United States']
excludeByTag=[name:en=Russia]
getGeodeskRelations=yes

//Get metropolitan France 
layerName=countries
layerGeometry=polygons
createLayer=yes
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=3][name:en='Metropolitan France']

//Get French dependencies which are level 3 and have an is in tag though none inside Europe
