//Defines the steps for creating an accessible atlas database 
database=land.db


//Get admin levels 2 to 6 from Europe 
//Starting with admin level 2 
//Exclude France and Russia
layer=countries
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=2]
geodeskAreaFilter=(-11.0,49.0,2.0,61.0)
excludeByTag=[name:en=France]
excludeByTag=[name:en=Russia]
getGeodeskRelations=yes

//Get metropolitan France 
layer=countries
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=3][name:en='Metropolitan France']
geodeskAreaFilter=(-11.0,49.0,2.0,61.0)
getGeodeskRelations=yes

//Get French dependencies which are level 3 and have an is in tag though none inside Europe 
layer=countries
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=3][is_in:country=France]
geodeskAreaFilter=(-11.0,49.0,2.0,61.0)
getGeodeskRelations=yes

//Get Russia as current planet latest 30th October 2025 leaves 2 line strings 
layer=countries
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=2][name:en=Russia]
getGeodeskRelations=yes

//Admin level 4 
layer=adminLevel4
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=4]
geodeskAreaFilter=(-11.0,49.0,2.0,61.0)
getGeodeskRelations=yes

//Admin level 5 
layer=adminLevel5
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=5]
geodeskAreaFilter=(-11.0,49.0,2.0,61.0)
getGeodeskRelations=yes

//Admin level 6 
layer=adminLevel6
geodeskSource=e:/osmdata/planet-latest.gol
geodeskFilter=a[type=boundary][boundary=administrative][admin_level=6]
geodeskAreaFilter=(-11.0,49.0,2.0,61.0)
getGeodeskRelations=yes
